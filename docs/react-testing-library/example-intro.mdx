import React , {useState} from  'react';
import 'bootstrap/dist/css/bootstrap.css';

function  App() {
    const [inputValue, setInputValue] = useState('');
    const [marque, setMarque]=useState('');
    const [date, setDate]=useState('');
    const [couleur, setCouleur]=useState('');

    const [showInputValue, setShowInputValue] = useState(false);
    const handleConfirmClick = () => {
        setShowInputValue(true);
    };
    // Function to handle input changes
    const handleInputChange = (event) => {
      setInputValue(event.target.value);
      
    };
    const hi = (event) =>{
        setMarque(event.target.value);
    }
    const h = (event) =>{
        setDate(event.target.value);
    }
    const i = (event) =>{
        setCouleur(event.target.value);
    }
  

    
    
    return(
        <div className="container">
            <h1>Gestion Voitures</h1>
            <label>Matricule:</label>
            <input type='text' placeholder='1-h-11111' onChange={handleInputChange}/>
            <label>Marque:</label>
            <select onChange={hi}>
                <option>Toyota</option>
                <option>Dacia</option>
                <option>Mercidice</option>
                <option>BMW</option>
            </select>
            <label>Date de mise en circulation:</label>
            <input type='date' placeholder='dd/mm/yyyy' onChange={h}/><br/>
            <label>Couleur</label>
            <input type="text" onChange={i}/><br/>
            <button onClick={handleConfirmClick} >Confirmer</button><br/>

            <h1>RÃ©capitulatif des informations:</h1>
            <ul >
                <li>Matricule:{inputValue}</li>
                <li>Marque:{marque}</li>
                <li>Date Mise en circulation:{date}</li>
                <li>Couleur:{couleur}</li>
            </ul>
        </div>
    )

};







export default App;
